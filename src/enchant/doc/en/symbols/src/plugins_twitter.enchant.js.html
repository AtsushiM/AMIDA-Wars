 45 /**
 46  * @scope enchant.nineleap.twitter.Game.prototype
 47  * enchant.Game or enchant.nineleap.Game クラスを拡張したクラス
 48  */
 49 enchant.nineleap.twitter.Game = enchant.Class.create(parentModule.Game, {
 50     initialize: function(width, height) {
 51         parentModule.Game.call(this, width, height);
 52         this._twitterRequests = [];
 53         this._twitterAssets = [];
 54         this.requireAuth = true;
 55         this.authorized = true;
 56         this.twitterQueue = 0;
 57     },
 58     start: function() {
 59         if (this.twitterQueue != 0) {
 60             if (this._twitterRequests.length) {
 61                 this.twitterAssets = {};
 62                 for (var i in this._twitterRequests) {
 63                     this.twitterAssets[this._twitterRequests[i].path] = new Array();
 64                 }
 65                 for (var i = 0, l = this._twitterRequests.length; i < l; i++) {
 66                     this._twitterRequests[i]._sendRequest();
 67                 }
 68             }
 69             return;
 70         }
 71         parentModule.Game.prototype.start.call(this);
 72     },
 73     /**
 74      * Twitterへのリクエストを送る。
 75      *
 76      * @param path {String} Twitter APIのパス
 77      * @param option {Object} Twitter APIへのリクエスト時に送信するオプション
 78      * @param checkError {Boolean} エラーで応答がない場合 (optional)
 79      */
 80     twitterRequest: function(path, option, checkError) {
 81         if (checkError == undefined) {
 82             checkError = true;
 83         }
 84         if (arguments.length == 2) {
 85             if (typeof arguments[1] == 'boolean') {
 86                 checkError = arguments[1];
 87                 option = {};
 88             } else {
 89                 checkError = true;
 90             }
 91         }
 92         this.twitterQueue++;
 93         var id = this._twitterRequests.length;
 94         var request = new enchant.nineleap.twitter.TwitterRequest(id, path, option, checkError);
 95         this._twitterRequests.push(request);
 96     },
 97 
 98     _setTwitterAssets: function(resBody, path, checkError) {
 99         var game = enchant.Game.instance;
100         if (!(resBody instanceof Array)) {
101             resBody = Array.prototype.slice.call(arguments);
102             resBody = resBody.filter(function(arg) {
103                 return (arg instanceof Object);
104             });
105         }
106         this.twitterQueue--;
107         if (resBody[0] == undefined) {
108             this.twitterAssets[path] = [];
109         } else if ('code' in resBody[0]) {
110             if (resBody[0].code == 401 && this.requireAuth) {
111                 window.location.replace('http://9leap.net/api/login?after_login=' + window.location.href);
112                 return ;
113             } else if (resBody[0].code == 401 && !this.requireAuth) {
114                 this.authorized = false;
115             } else if (checkError) {
116                 alert (resBody[0].code + ' Error' + '\n Please try reload');
117                 throw new Error(resBody[0].code + ': ' +resBody[0].error);
118             } else {
119                 this.twitterAssets[path] = [];
120             }
121         } else {
122             for (var i = 0, l = resBody.length; i < l; i++) {
123                 if ('name' in resBody[i]) {
124                     this.twitterAssets[path][i] = new enchant.nineleap.twitter.TwitterUserData(resBody[i]);
125                     this._twitterAssets.push(resBody[i]['profile_image_url']);
126                     if ('status' in resBody[i]) {
127                         this.twitterAssets[path][i].status = new enchant.nineleap.twitter.TwitterStatusData(resBody[i].status);
128                     }
129                 } else {
130                     this.twitterAssets[path][i] = new enchant.nineleap.twitter.TwitterStatusData(resBody[i]);
131                     this.twitterAssets[path][i].user = new enchant.nineleap.twitter.TwitterUserData(resBody[i].user);
132                     this._twitterAssets.push(resBody[i].user['profile_image_url']);
133                 }
134             }
135         }
136         if (this.twitterQueue == 0) {
137             this.start();
138         }
139     }

142 enchant.nineleap.twitter.TwitterRequest = enchant.Class.create({
143     initialize: function(id, path, option, checkError) {
144         this.path = path;
145         this.checkError = checkError;
146         var callback = '?callback=enchant.Game.instance._twitterRequests[' + id + ']._callback';
147         var src = 'http://9leap.net/api/twitter/' + path + '.json' + callback;
148         if (option) {
149             for (var key in option) {
150                 src += '&' + key + '=' + option[key];
151             }
152         }
153         this.script = document.createElement('script');
154         this.script.type = 'text/javascript';
155         this.script.src = src;
156     },
157     _callback: function(resBody) {
158         enchant.Game.instance._setTwitterAssets(resBody, this.path, this.checkError);
159     },
160     _sendRequest: function() {
161         document.head.appendChild(this.script);
162     }
163 });

174             var height = 48;
175         }
176         var g = enchant.Game.instance;
177         var sp = new Sprite(width, height);
178         sp.image = g.assets[this['profile_image_url']];
179         return sp;
180     }
