 32 /**
 33  * @scope enchant.CanvasGroup
 34  */
 35 enchant.CanvasGroup = enchant.Class.create(enchant.Group,{
 36 	initialize:function(){
 37 		enchant.Group.call(this);
 38 		
 39 		cashing = function(node){
 40 		   var row = node._image.width / node.width | 0;
 41 			node._sx = 	~~(node.width*(node.frame%row));
 42 			node._sy = 	~~((node.frame/row|0)*node.height);
 43 			node._sw = 	~~node._width;
 44 			node._sh = 	~~node._height;
 45 			node._dx = 	~~(node.width*(node.scaleX-1)*0.5);
 46 			node._dy = 	~~(node.height*(node.scaleY-1)*0.5);
 47 			node._dw = 	~~(node.width*node.scaleX);
 48 			node._dh = 	~~(node.height*node.scaleY);
 49 			if((node._image.width==node.width)&&(node.scaleX==1&&node.scaleY==1)){
 50 				node.draw = function(context){
 51 						context.drawImage(this.image._element,
 52 							this.x+this._dx,
 53 							this.y+this._dy);					
 54 				}
 55 			}else{
 56 				node.draw = function(context){
 57 					context.drawImage(this.image._element,
 58 						this._sx,this._sy,
 59 						this._sw,this._sh,
 60 						this.x+this._dx,
 61 						this.y+this._dy,
 62 						this._dw,this._dh);	
 63 				}
 64 			}
 65 		};
 66 		
 67 		this.addEventListener('enterframe',this.enterframe=function(){
 68 			this.context.clearRect(0,0,game.width,game.height);
 69 			var context = this.context;
 70 			var nodes = this.canvasChildNodes;
 71 			var copmpsite = "source-atob"
 72 			var fiz=0;
 73 			for(i=0,len =  nodes.length ;i<len;i++){
 74 				var node = nodes[i];
 75 				if(node.alphaBlending){
 76 					context.globalCompositeOperation = 
 77 															node.alphaBlending;
 78 				}else{
 79 					context.globalCompositeOperation = 
 80 															"source-atob";
 81 				}
 82 				node.draw(context);
 83 			}
 84 		});
 85 
 86         this._element = document.createElement('canvas');
 87         this._element.width = game.width;
 88         this._element.height = game.height;
 89         this._element.style.position = 'absolute';
 90         this.context = this._element.getContext('2d');
 91 
 92 		var node=this;
 93         this.canvasChildNodes = [];
 94         this.currentTime = Date.now();
 95 
 96 		this.addChild = function(node){
 97 			cashing(node);
 98 			node._listeners.render=[];
 99 	        this.childNodes[this.childNodes]=node;
100 	        this.canvasChildNodes[this.canvasChildNodes.length]=node;
101 	        node.parentNode = this;
102 	        node.dispatchEvent(new enchant.Event('added'));
103 	        if (this.scene) {
104 	            var e = new enchant.Event('addedtoscene');
105 	            node.scene = this.scene;
106 	            node.dispatchEvent(e);
107 	            node._updateCoordinate();
108 	        }	        
109 		};
110 	},
111 	removeChild:function(node){
112 		for(i = 0, len= this.canvasChildNodes.length; i < len; i++){
113 	 		if(this.canvasChildNodes[i] === node){
114  				this.canvasChildNodes.splice(i,1);
115 			}
116 		}
117 	}
118 });
