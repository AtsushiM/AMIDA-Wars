 44  * 方向キーパッドのクラス: Pad

 47 enchant.ui.Pad = enchant.Class.create(enchant.Sprite, {
 48     /**
 49      * 方向キーパッドオブジェクトを作成する。
 50      * @constructs
 51      * @extends enchant.Sprite
 52      */
 53     initialize: function() {
 54         var game = enchant.Game.instance;
 55         var image = game.assets['pad.png'];
 56         enchant.Sprite.call(this, image.width / 2, image.height);
 57         this.image = image;
 58         this.input = { left: false, right: false, up: false, down:false };
 59         this.addEventListener('touchstart', function(e) {
 60             this._updateInput(this._detectInput(e.localX, e.localY));
 61         });
 62         this.addEventListener('touchmove', function(e) {
 63             this._updateInput(this._detectInput(e.localX, e.localY));
 64         });
 65         this.addEventListener('touchend', function(e) {
 66             this._updateInput({ left: false, right: false, up: false, down:false });
 67         });
 68     },
 69     _detectInput: function(x, y) {
 70         x -= this.width / 2;
 71         y -= this.height / 2;
 72         var input = { left: false, right: false, up: false, down:false };
 73         if (x * x + y * y > 200) {
 74             if (x < 0 && y < x * x * 0.1 && y > x * x * -0.1) {
 75                 input.left = true;
 76             }
 77             if (x > 0 && y < x * x * 0.1 && y > x * x * -0.1) {
 78                 input.right = true;
 79             }
 80             if (y < 0 && x < y * y * 0.1 && x > y * y * -0.1) {
 81                 input.up = true;
 82             }
 83             if (y > 0 && x < y * y * 0.1 && x > y * y * -0.1) {
 84                 input.down = true;
 85             }
 86         }
 87         return input;
 88     },
 89     _updateInput: function(input) {
 90         var game = enchant.Game.instance;
 91         ['left', 'right', 'up', 'down'].forEach(function(type) {
 92             if (this.input[type] && !input[type]) {
 93                 game.dispatchEvent(new Event(type + 'buttonup'));
 94             }
 95             if (!this.input[type] && input[type]) {
 96                 game.dispatchEvent(new Event(type + 'buttondown'));
 97             }
 98         }, this);
 99         this.input = input;
100     }
101 });

103 /**

106  */
107 enchant.ui.APad = enchant.Class.create(enchant.Group, {
108     /**
109      * アナログパッドオブジェクトを作成する。
110      * @constructs
111      * @extends enchant.Group
112      * @param mode
113      *   'direct': 入力ベクトルは正規化されない (大きさは 0~1 の間)
114      *   'normal': 入力ベクトルを常に正規化する (大きさは常に1となる)
115      */
116 	initialize: function(mode) {
117         var game = enchant.Game.instance;
118         var image = game.assets['apad.png'];
119 		var w = this.width = image.width;
120 		var h = this.height = image.height;
121         enchant.Group.call(this);
122 
123 		this.outside = new Sprite(w, h);
124 		var outsideImage = new Surface(w, h);
125 		outsideImage.draw(image,     0,     0,   w, h/4,     0,     0,   w, h/4);
126 		outsideImage.draw(image,     0, h/4*3,   w, h/4,     0, h/4*3,   w, h/4);
127 		outsideImage.draw(image,     0,   h/4, w/4, h/2,     0,   h/4, w/4, h/2);
128 		outsideImage.draw(image, w/4*3,   h/4, w/4, h/2, w/4*3,   h/4, w/4, h/2);
129 		this.outside.image = outsideImage;
130 		this.inside = new Sprite(w/2, h/2);
131 		var insideImage = new Surface(w/2, h/2);
132 		insideImage.draw(image, w/4, h/4, w/2, h/2, 0, 0, w/2, h/2);
133 		this.inside.image = insideImage;
134         this.r = w/2;
135 
136         /**
137          * isTouched
138          * @type {Boolean}
139          * タッチされているかどうか
140          */
141 		this.isTouched = false;
142 
143         /**
144          * vx, vy
145          * @type {Number}
146          * 入力ベクトルの(x, y)方向の大きさ
147          */
148         this.vx = 0;
149 		this.vy = 0;
150 
151         /**
152          * rad, dist
153          * @type {Number}
154          * 入力ベクトルの極座標表示
155          * radは角度、distはベクトルの大きさを示す
156          */
157         this.rad = 0;
158         this.dist = 0;
159 
160 		if (mode === 'direct') {
161 			this.mode = 'direct';
162 		} else {
163 			this.mode = 'normal';
164 		}
165         this._updateImage();
166 		this.addChild(this.inside);
167 		this.addChild(this.outside);
168 		this.addEventListener('touchstart', function(e) {
169 			this._detectInput(e.localX, e.localY);
170 			this._calcPolar(e.localX, e.localY);
171 			this._updateImage(e.localX, e.localY);
172 			this._dispatchPadEvent('apadstart');
173 			this.isTouched = true;
174 		});
175 		this.addEventListener('touchmove', function(e) {
176 			this._detectInput(e.localX, e.localY);
177 			this._calcPolar(e.localX, e.localY);
178 			this._updateImage(e.localX, e.localY);
179 			this._dispatchPadEvent('apadmove');
180 		});
181 		this.addEventListener('touchend', function(e) {
182 			this._detectInput(this.width / 2, this.height / 2);
183 			this._calcPolar(this.width / 2, this.height / 2);
184 			this._updateImage(this.width / 2, this.height / 2);
185 			this._dispatchPadEvent('apadend');
186 			this.isTouched = false;
187 		});
188 	},
189 	_dispatchPadEvent: function(type) {
190 		var e = new Event(type);
191 		e.vx = this.vx;
192 		e.vy = this.vy;
193 		e.rad = this.rad;
194 		e.dist = this.dist;
195 		this.dispatchEvent(e);
196 	},
197 	_updateImage: function(x, y) {
198 		x -= this.width / 2;
199 		y -= this.height / 2;
200 		this.inside.x = this.vx * (this.r - 10) + 25;
201 		this.inside.y = this.vy * (this.r - 10) + 25;
202 	},
203 	_detectInput: function(x, y) {
204 		x -= this.width / 2;
205 		y -= this.height / 2;
206 		var distance = Math.sqrt(x * x + y * y);
207 		var tan = y / x;
208 		var rad = Math.atan(tan);
209 		var dir = x / Math.abs(x);
210 		if (distance == 0) {
211 			this.vx = 0;
212 			this.vy = 0;
213 		} else if (x == 0) {
214 			this.vx = 0;
215 			if (this.mode == 'direct') {
216 				this.vy = (y / this.r);
217 			} else {
218 				dir = y / Math.abs(y);
219 				this.vy = Math.pow((y / this.r), 2) * dir;
220 			}
221 		} else if (distance < this.r) {
222 			if (this.mode == 'direct') {
223 				this.vx = (x / this.r);
224 				this.vy = (y / this.r);
225 			} else {
226 				this.vx = Math.pow((distance / this.r), 2) * Math.cos(rad) * dir;
227 				this.vy = Math.pow((distance / this.r), 2) * Math.sin(rad) * dir;
228 			}
229 		} else {
230 			this.vx = Math.cos(rad) * dir;
231 			this.vy = Math.sin(rad) * dir;
232 		}
233 	},
234 	_calcPolar: function(x, y) {
235 		x -= this.width / 2;
236 		y -= this.height / 2;
237 		var add = 0;
238 		var rad = 0;
239 		var dist = Math.sqrt(x * x + y * y);
240 		if (dist > this.r) {
241 			dist = this.r;
242 		}
243 		dist /= this.r;
244 		if (this.mode == 'normal') {
245 			dist *= dist;
246 		}
247 		if (x >= 0 && y < 0) {
248 			add = Math.PI/2*3;
249 			rad = x / y;
250 		} else if (x < 0 && y <= 0) {
251 			add = Math.PI;
252 			rad = y / x;
253 		} else if (x <= 0 && y > 0) {
254 			add = Math.PI / 2;
255 			rad = x / y;
256 		} else if (x > 0 && y >= 0) {
257 			add = 0;
258 			rad = y / x;
259 		} 
260 		if (x == 0 || y == 0) {
261 			rad = 0;
262 		}
263 		this.rad = Math.abs(Math.atan(rad)) + add; 
264 		this.dist = dist;
265 	}
266 });

270  * @scope enchant.ui.Button.prototype
271  */
272 enchant.ui.Button = enchant.Class.create(enchant.Entity, {
273     /**
274      * ボタンオブジェクトを作成する。
275      * @constructs
276      * @extends enchant.Entity
277      */
278     initialize: function(text, theme, height, width) {
279         enchant.Entity.call(this);
280         
281         // ベンタプレフィクスを判定
282         this.VENDOR_PREFIX = (function() {
283             var ua = navigator.userAgent;
284             if (ua.indexOf('Opera') != -1) {
285                 return '-O-';
286             } else if (ua.indexOf('MSIE') != -1) {
287                 return '-ms-';
288             } else if (ua.indexOf('WebKit') != -1) {
289                 return '-webkit-';
290             } else if (navigator.product == 'Gecko') {
291                 return '-Moz-';
292             } else {
293                 return '';
294             }
295         })();
296 
297         this.width = width || null;
298         this.height = height || null;
299         this.text = text;
300         this.pressed = false;
301 
302         // デフォルトのスタイル (テーマで上書き可能)
303         var style = this._element.style;
304         style["display"]          = "inline-block";
305         style["font-size"]        = "12px";
306         style["height"]           = "2em";
307         style["line-height"]      = "2em";
308         style["min-width"]        = "2em";
309         style["padding"]          = "2px 10px";
310         style["text-align"]       = "center";
311         style["font-weight"]      = "bold";
312         style[this.VENDOR_PREFIX + "border-radius"] = "0.5em";
313 
314         // テーマの指定がなければ "dark" を使う
315         theme = theme || "dark";
316 
317         if(typeof theme === "string"){
318             // theme 引数が string なら、その名前のデフォルトテーマを使う
319             this.theme = enchant.ui.Button.DEFAULT_THEME[theme];
320         }else{
321             // theme 引数が object なら、その引数をテーマとして扱う
322             this.theme = theme;
323         }
324 
325         // テーマを適用する
326         this._applyTheme(this.theme.normal);
327 
328         // タッチしたときの挙動
329         this.addEventListener("touchstart", function(){
330             this._applyTheme(this.theme.active);
331             this.pressed = true;
332             this.y ++;
333         });
334 
335         // タッチが離されたときの挙動
336         this.addEventListener("touchend", function(){
337             this._applyTheme(this.theme.normal);
338             this.pressed = false;
339             this.y --;
340         });
341     },
342     _applyTheme: function(theme){
343         var style = this._element.style;
344         for(var i in theme) if(theme.hasOwnProperty(i)){
345             var prop = i.replace(/\-VENDOR\-/, this.VENDOR_PREFIX);
346             style[prop] = theme[i].replace(/\-VENDOR\-/, this.VENDOR_PREFIX);
347             console.log("apply theme",i,":",theme[i]);
348             console.log("# vendor prefix replaced : ",prop,":", theme[i].replace(/\-VENDOR\-/, this.VENDOR_PREFIX));
349         }
350     },
351     /**
352      * 表示するテキスト
353      * @type {String}
354      */
355     text: {
356         get: function() {
357             return this._element.innerHTML;
358         },
359         set: function(text) {
360             this._element.innerHTML = text;
361         }
362     },
363     /**
364      * フォントサイズ
365      */
366     size: {
367         get: function() {
368             return this._element.style["fontSize"];
369         },
370         set: function(size){
371             this._element.style["fontSize"] = size;
372         }
373     },
374     /**
375      * フォントの指定
376      * @type {String}
377      */
378     font: {
379         get: function() {
380             return this._style.font;
381         },
382         set: function(font) {
383             this._style.font = font;
384         }
385     },
386     /**
387      * Text color settings.
388      * CSS��'color' Can be set to same format as properties.
389      * @type {String}
390      */
391     color: {
392         get: function() {
393             return this._style.color;
394         },
395         set: function(color) {
396             this._style.color = color;
397         }
398     }
399 });
