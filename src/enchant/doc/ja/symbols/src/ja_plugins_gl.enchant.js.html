 41 function parseColor(arg) {
 42     var color = [];
 43     if (typeof arg == 'string') {
 44         if (arg.match(/#/)) {
 45             arg.match(/[0-9a-fA-F]{2}/g).forEach(function(n) {
 46                     color[color.length] = ('0x' + n - 0) / 255;
 47                     });
 48             color[color.length] = 1.0;
 49         } else if (arg.match(/rgba/)) {
 50             arg.match(/[0-9]{1,3},/g).forEach(function(n) {
 51                     color[color.length] = parseInt(n, 10) / 255;
 52                     });
 53             color[color.length] = parseFloat(arg.match(/[0-9]\.[0-9]{1,}/)[0]);
 54         } else if (arg.match(/rgb/)) {
 55             arg.match(/[0-9]{1,3},/g).forEach(function(n) {
 56                     color[color.length] = parseInt(n, 10) / 255;
 57                     });
 58             color[color.length] = 1.0;
 59         }
 60     } else if (arg instanceof Array) {
 61         color = arg;
 62     }
 63     return color;
 64 };

 67 function createDebugContext(context) {
 68     var ctx = {};
 69     var type = '';
 70     context.names = {};
 71     for (prop in context) {
 72         type = typeof context[prop];
 73         if (type == 'function') {
 74             ctx[prop] = (function(context, prop) {
 75                 return function() {
 76                     value = undefined;
 77                     value = context[prop].apply(context, arguments);
 78                     error = context.getError();
 79                     if (error) {
 80                         if (prop != 'viewport') {
 81                             console.log(context.names[error] + '[' + error + ']: ' + prop);
 82                             console.log(arguments);
 83                         }
 84                     }
 85                     return value;
 86                 }
 87             })(context, prop);
 88         } else if (type == 'number') {
 89             context.names[context[prop]] = prop;
 90             ctx[prop] = context[prop];
 91         } else {
 92             ctx[prop] = context[prop];
 93         }
 94     }
 95     ctx.getPropName = function(index) {
 96         return context.names[index];
 97     };
 98     return ctx;

138     var point2point = function(p1, p2) {
139         var vx = p1.x + p1.parent.x - p2.x - p2.parent.x;
140         var vy = p1.y + p1.parent.y - p2.y - p2.parent.y;
141         var vz = p1.z + p1.parent.z - p2.z - p2.parent.z;
142         return (vx * vx + vy * vy + vz * vz);
143     };
144     var point2BS = function(p, bs) {
145         return (point2point(p, bs) - bs.radius * bs.radius);
146     };
147     var point2AABB = function(p, aabb) {
148         var ppx = p.x + p.parent.x;
149         var ppy = p.y + p.parent.y;
150         var ppz = p.z + p.parent.z;
151         var px = aabb.parent.x + aabb.x + aabb.scale;
152         var py = aabb.parent.y + aabb.y + aabb.scale;
153         var pz = aabb.parent.z + aabb.z + aabb.scale;
154         var nx = aabb.parent.x + (aabb.x - aabb.scale);
155         var ny = aabb.parent.y + (aabb.y - aabb.scale);
156         var nz = aabb.parent.z + (aabb.z - aabb.scale);
157         var dist = 0;
158         if (ppx < nx) {
159             dist += (ppx - nx) * (ppx - nx);
160         } else if (px < ppx) {
161             dist += (ppx - px) * (ppx - px);
162         }
163         if (ppy < ny) {
164             dist += (ppy - ny) * (ppy - ny);
165         } else if (py < ppy) {
166             dist += (ppy - py) * (ppy - py);
167         }
168         if (ppz < nz) {
169             dist += (ppz - nz) * (ppz - nz);
170         } else if (pz < ppz) {
171             dist += (ppz - pz) * (ppz - pz);
172         }
173         return dist;
174     };
175     var point2OBB = function(p, obb) {
176         return 1;
177     };
178     var BS2BS = function(bs1, bs2) {
179         return (point2point(bs1, bs2) - (bs1.radius + bs2.radius) * (bs1.radius + bs2.radius));
180     };
181     var BS2AABB = function(bs, aabb) {
182         return (point2AABB(bs, aabb) - bs.radius * bs.radius);
183     };

185         return 1;
