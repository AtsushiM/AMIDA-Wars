1 /**
  2  * nineleap.enchant.js
  3  * @version 0.3 (2012/02/01)
  4  * @requires enchant.js v0.4.0 or later
  5  *
  6  * @description
  7  * enchant.js extension for 9leap.net
  8  * 9leap.net 向けの enchant.js 拡張プラグイン。
  9  * game.end の引数にスコアと結果の文字列を渡すことで、ランキングに登録できる。
 10  * (9leapにアップロードした後のみランキング画面にジャンプする)
 11  *
 12  * @usage
 13 
 14  var game = new Game(320, 320);
 15 
 16  game.onload = function(){
 17      // executed after player pushed "START"
 18      // ...
 19      if(some.condition)game.end(score, result);
 20  };
 21 
 22  game.start();
 23

 26 (function () {
 27 
 28 enchant.nineleap = { assets: ['start.png', 'end.png'] };
 29 
 30 /**
 31  * @scope enchant.nineleap.Game.prototype
 32  */
 33 enchant.nineleap.Game = enchant.Class.create(enchant.Game, {
 34     /**
 35      *
 36      * @param width
 37      * @param height
 38      */
 39     initialize: function(width, height) {
 40         enchant.Game.call(this, width, height);
 41         this.addEventListener('load', function() {
 42             var game = this;
 43             this.startScene = new SplashScene();
 44             this.startScene._element.style.zIndex = 10;
 45             this.startScene.image = this.assets['start.png'];
 46             this.startScene.addEventListener('touchend', function() {
 47                 if (game.started == false) {
 48                     if (game.onstart != null) game.onstart();
 49                     game.started = true;
 50                     gameStart = true;   // deprecated
 51                 }
 52                 if (game.currentScene == this) game.popScene();
 53                 this.removeEventListener('touchend', arguments.callee);
 54             });
 55             this.addEventListener('keydown', function() {
 56                 if (this.started == false) {
 57                     if(this.onstart != null) this.onstart();
 58                     this.started = true;
 59                     gameStart = true;   // deprecated
 60                 }
 61                 if (game.currentScene == game.startScene) game.popScene();
 62                 this.removeEventListener('keydown', arguments.callee);
 63             });
 64             this.pushScene(this.startScene);
 65 
 66             this.endScene = new SplashScene();
 67             this.endScene.image = this.assets['end.png'];
 68         });
 69         this.scoreQueue = false;
 70         this.started = false;
 71         gameStart = false; // deprecated
 72     },
 73 
 74     loadImage: function(src, callback) {
 75         if (callback == null) callback = function() {};
 76         this.assets[src] = enchant.Surface.load(src);
 77         this.assets[src].addEventListener('load', callback);
 78     },
 79 
 80     start: function() {
 81         var game = this;
 82 
 83         if (this._intervalID) {
 84             window.clearInterval(this._intervalID);
 85         } else if (this._assets.length) {
 86             var o = {};
 87             var assets = this._assets.filter(function(asset) {
 88                 return asset in o ? false : o[asset] = true;
 89             });
 90             var tAssets = (this._twitterAssets != undefined)? this._twitterAssets : [];
 91             var nAssets = (this._netpriceData != undefined)? this._netpriceData : [];
 92             var mAssets = (this._memoryAssets != undefined) ? this._memoryAssets : [];
 93             var loaded = 0;
 94             var total = assets.length + tAssets.length + nAssets.length + mAssets.length;
 95             for (var i = 0, len = assets.length; i < len; i++) {
 96                 this.load(assets[i], function() {
 97                     var e = new enchant.Event('progress');
 98                     e.loaded = ++loaded;
 99                     e.total = total;
100                     game.dispatchEvent(e);
101                     if (loaded == total) {
102                         game.removeScene(game.loadingScene);
103                         game.dispatchEvent(new enchant.Event('load'));
104                     }
105                 });
106             }
107             for (var i = 0, len = tAssets.length; i < len; i++) {
108                 this.loadImage(tAssets[i], function() {
109                     var e = new enchant.Event('progress');
110                     e.loaded = ++loaded;
111                     e.total = total;
112                     game.dispatchEvent(e);
113                     if (loaded == total) {
114                         game.removeScene(game.loadingScene);
115                         game.dispatchEvent(new enchant.Event('load'));
116                     }
117                 });
118             }
119             for (var i = 0, len = mAssets.length; i < len; i++) {
120                 this.loadImage(mAssets[i], function() {
121                     var e = new enchant.Event('progress');
122                     e.loaded = ++loaded;
123                     e.total = total;
124                     game.dispatchEvent(e);
125                     if (loaded == total) {
126                         game.removeScene(game.loadingScene);
127                         game.dispatchEvent(new enchant.Event('load'));
128                     }
129                 });
130             }
131 
132             for (var i = 0, len = nAssets.length; i < len; i++) {
133                 this.loadImage(nAssets[i], function() {
134                     var e = new enchant.Event('progress');
135                     e.loaded = ++loaded;
136                     e.total = total;
137                     game.dispatchEvent(e);
138                     if (loaded == total) {
139                         game.removeScene(game.loadingScene);
140                         game.dispatchEvent(new enchant.Event('load'));
141                     }
142                 });
143             }
144             this.pushScene(this.loadingScene);
145         } else {
146             this.dispatchEvent(new enchant.Event('load'));
147         }
148         this.currentTime = Date.now();
149         this._intervalID = window.setInterval(function() {
150             game._tick()
151         }, 1000 / this.fps);
152         this.running = true;
153     },
154 
155     end: function(score, result, img) {
156         if(img !== undefined){
157             this.endScene.image = img;
158         }
159         this.pushScene(this.endScene);
160         if (location.hostname == 'r.jsgames.jp') {
161             var submit = function() {
162                 var id = location.pathname.match(/^\/games\/(\d+)/)[1]; 
163                 location.replace([
164                     'http://9leap.net/games/', id, '/result',
165                     '?score=', encodeURIComponent(score),
166                     '&result=', encodeURIComponent(result)
167                 ].join(''));
168             }
169             this.endScene.addEventListener('touchend', submit);
170             window.setTimeout(submit, 3000);
171         }
172         enchant.Game.instance.end = function(){ };
173     }
174 });

176 /**
177  * @scope enchant.nineleap.SplashScene.prototype
178  */
179 enchant.nineleap.SplashScene = enchant.Class.create(enchant.Scene, {
180     image: {
181         get: function() {
182             return this._image;
183         },
184         set: function(image) {
185             this._image = image;
186 
187             while (this.firstChild) {
188                 this.removeChild(this.firstChild);
189             }
190             var sprite = new Sprite(image.width, image.height);
191             sprite.image = image;
192             sprite.x = (this.width - image.width) / 2;
193             sprite.y = (this.height - image.height) / 2;
194             this.addChild(sprite);
195         }
196     }
197 });
